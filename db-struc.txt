DELIMITER $$

CREATE TRIGGER trg_stock_mov_after_insert
AFTER INSERT ON stock_mov
FOR EACH ROW
BEGIN

    DECLARE stock_qty INT;

    -- Calculate inventory balance
    SELECT
        IFNULL(SUM(CASE WHEN movement_type='IN' THEN quantity ELSE 0 END),0) -
        IFNULL(SUM(CASE WHEN movement_type='OUT' THEN quantity ELSE 0 END),0)
    INTO stock_qty
    FROM stock_mov
    WHERE product_id = NEW.product_id;

    -- Update inventory stock table
    UPDATE inventory_level
    SET quantity = stock_qty
    WHERE product_id = NEW.product_id;

END$$

DELIMITER ;

	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	emp_id INT,
	date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	basic_salary DECIMAL(10,2) NOT NULL,
	deductions DECIMAL(10,2) DEFAULT 0,
	allowance DECIMAL(10,2) DEFAULT 0,
	overtime DECIMAL(10,2) DEFAULT 0,
	net_salary DECIMAL(10,2) GENERATED ALWAYS AS (
        basic_salary + allowance + overtime - deductions
    	) STORED,
	FOREIGN KEY (emp_id) REFERENCES employees(id)
);
